//1. Encontrar el digito mayor de un numero.
#include "stdafx.h"
#include <iostream>
using namespace std;

int mayor(int n)
{
   if(n==0)
     return 0;
   else {
           int digMayor=abs(n)%10;
	   if(digMayor<mayor(abs(n)/10))
	      return mayor(abs(n)/10);
	     else return digMayor;
	}
}

{
    int n;
    cout<<"Ingrese el numero"<<endl;
